{# A badge in a list or grid context -- requires "item" and optionally "url" in parent template #} {# itemUrl simplifies getting the URL for the badge: it can be set in the parent template overriding the badge (or in some cases supplying it if it doesn't have a URL #} {% macro itemUrl(url, item) %} {% if url %} {{ url }} {% else %} {{ item.url }} {% endif %} {% endmacro %}
